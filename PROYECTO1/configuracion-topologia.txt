
# Configurando el servidor (SW1)
enable
	configure terminal
	vtp mode server
	
	vtp domain P31
	vtp password usac
	vtp version 2
	exit
	wr

# Ver estado del vtp
show vtp status

# Configurando modo clientes
enable
	configure terminal
	vtp mode client

	vtp domain P31
	vtp password usac
	exit
	wr
	
	show vtp status

# Establecer en modo trunkal menos el SW9
enable
	configure terminal
	interface range f0/rango
	switchport trunk encapsulation dot1q # Para los de capa 3
	switchport mode trunk
	switchport trunk allowed vlan all
	exit
	exit
	wr
	
	show startup-config
